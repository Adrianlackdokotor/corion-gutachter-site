### 2) GA4 tracking (Measurement ID: G-RYF77GKESM)
GA4 should respect cookie consent (EU/Germany):
- On page load: set consent denied and do not send events.
- When user clicks “Akzeptieren” on cookie banner: load GA script + set consent granted + send pageview.
- If “Ablehnen”: do nothing.

Add to the main HTML template/head (so it exists on all pages: DE + RO), and ensure it loads only once:
<!-- Google tag (gtag.js) -->
<script async src="https://www.googletagmanager.com/gtag/js?id=G-RYF77GKESM"></script>
<script>
  window.dataLayer = window.dataLayer || [];
  function gtag(){dataLayer.push(arguments);}

  // Default: consent denied until user accepts cookies
  gtag('consent', 'default', {
    'ad_storage': 'denied',
    'analytics_storage': 'denied',
    'functionality_storage': 'denied',
    'personalization_storage': 'denied',
    'security_storage': 'granted'
  });

  function initializeGA(){
    // set consent granted only after user accepts cookies
    gtag('consent', 'update', {
      'analytics_storage': 'granted'
    });
    gtag('js', new Date());
    gtag('config', 'G-RYF77GKESM', { 'anonymize_ip': true });
  }

  // generic helper used by existing tracking hooks
  window.trackGaEvent = function(eventName, params = {}){
    // only fire after GA is initialized and with consent granted
    if (!window.trackGaEventInitialized) return;
    gtag('event', eventName, {
      page_path: location.pathname,
      ...params
    });
  };

  // register init flag once GA is initialized
  window.trackGaEventInitialized = false;
  window.initializeAndMarkGA = function(){
    initializeGA();
    window.trackGaEventInitialized = true;
  }
</script>
